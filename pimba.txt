TITLE "random";
SUBDESIGN random
(
    clock : INPUT;
    led[4..1]        : OUTPUT;
    random_out[1..0] : OUTPUT
)

VARIABLE
    ----------------------------------------------------------------
    -- DIVISOR DE CLOCK
    ----------------------------------------------------------------
    counter[25..0] : DFF;
    ----------------------------------------------------------------
    -- LFSR (Sorteador Pseudo-Aleatório)
    ----------------------------------------------------------------
    lfsr[15..0] : DFF;

BEGIN
    ----------------------------------------------------------------
    -- DIVISOR DE CLOCK
    ----------------------------------------------------------------
    counter[].clk = clock;
    counter[].d = counter[].q + 1;
    
    ----------------------------------------------------------------
    -- CLOCK DO LFSR CORRIGIDO
    -- Mudado de counter[23].q para clock (50 MHz) para acompanhar
    -- a velocidade de gravação da memória no estado EST_IDLE.
    ----------------------------------------------------------------
    lfsr[].clk = clock;
    
    ----------------------------------------------------------------
    -- SAÍDA PSEUDO-ALEATÓRIA
    ----------------------------------------------------------------
    random_out[] = lfsr[1..0].q;
    
    ----------------------------------------------------------------
    -- SEED INICIAL (Garante que o LFSR não comece zerado)
    ----------------------------------------------------------------
    IF lfsr[].q == B"0000000000000000" THEN

        lfsr[].d = B"0000000000000001";

    ELSE
        ------------------------------------------------------------
        -- LFSR 16 BITS (Rotaciona os bits a cada ciclo)
        ------------------------------------------------------------
        lfsr[15].d = lfsr[14].q;
        lfsr[14].d = lfsr[13].q;
        lfsr[13].d = lfsr[12].q;
        lfsr[12].d = lfsr[11].q;
        lfsr[11].d = lfsr[10].q;
        lfsr[10].d = lfsr[9].q;
        lfsr[9].d  = lfsr[8].q;
        lfsr[8].d  = lfsr[7].q;
        lfsr[7].d  = lfsr[6].q;
        lfsr[6].d  = lfsr[5].q;
        lfsr[5].d  = lfsr[4].q;
        lfsr[4].d  = lfsr[3].q;
        lfsr[3].d  = lfsr[2].q;
        lfsr[2].d  = lfsr[1].q;
        lfsr[1].d  = lfsr[0].q;

        -- Equação de realimentação (polinômio de taps)
        lfsr[0].d =
            lfsr[15].q $
            lfsr[13].q $
            lfsr[12].q $
            lfsr[10].q;

    END IF;

    ----------------------------------------------------------------
    -- CONTROLE DOS LEDs (Efeito visual de oscilação rápida)
    ----------------------------------------------------------------
    IF counter[23].q THEN

        CASE lfsr[1..0].q IS

            WHEN B"00" =>
                led[] = B"1110";

            WHEN B"01" =>
                led[] = B"1101";

            WHEN B"10" =>
                led[] = B"1011";

            WHEN B"11" =>
                led[] = B"0111";

        END CASE;

    ELSE

        led[] = B"1111";

    END IF;

END;
